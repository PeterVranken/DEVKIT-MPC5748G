689 static ALWAYS_INLINE void rtos_osSuspendAllInterrupts(void)
690 {
691  /* The completion synchronizing character of the wrteei instruction forms the memory
692  barrier, which ensures that all memory operations before the now entered critical
693  section are completed before we enter (see core RM, 4.6.1, p. 151). The "memory"
694  constraint ensures that the compiler won't reorder instructions from behind the
695  wrteei to before it. */
696  asm volatile ( /* AssemblerTemplate */
697  "wrteei 0\n"
698  : /* OutputOperands */
699  : /* InputOperands */
