825static ALWAYS_INLINE void rtos_osSuspendAllInterrupts(void)
826{
827 /* The completion synchronizing character of the wrteei instruction forms the memory
828 barrier, which ensures that all memory operations before the now entered critical
829 section are completed before we enter (see core RM, 4.6.1, p. 151). The "memory"
830 constraint ensures that the compiler won't reorder instructions from behind the
831 wrteei to before it. */
832 asm volatile ( /* AssemblerTemplate */
833 "wrteei 0\n"
834 : /* OutputOperands */
835 : /* InputOperands */
836 : /* Clobbers */ "memory"
837 );
