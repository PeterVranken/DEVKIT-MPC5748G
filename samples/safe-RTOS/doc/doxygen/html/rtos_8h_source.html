762static ALWAYS_INLINE void rtos_osSuspendAllInterrupts(void)
763{
764 /* The completion synchronizing character of the wrteei instruction forms the memory
765 barrier, which ensures that all memory operations before the now entered critical
766 section are completed before we enter (see core RM, 4.6.1, p. 151). The "memory"
767 constraint ensures that the compiler won't reorder instructions from behind the
768 wrteei to before it. */
769 asm volatile ( /* AssemblerTemplate */
770 "wrteei 0\n"
771 : /* OutputOperands */
772 : /* InputOperands */
773 : /* Clobbers */ "memory"
774 );
775} /* End of rtos_osSuspendAllInterrupts */
